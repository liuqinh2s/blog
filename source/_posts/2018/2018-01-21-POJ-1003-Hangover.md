这一题 AC 的速度还是挺快的，题目本身也是很简单：

C++ 代码：

```C++
#include <iostream>
#include <string>
#include <sstream>
using namespace std;

string double2Str(const double &double_temp){
    stringstream ss;
    ss << double_temp;
    string str;
    ss >> str;
    return str;
}

int main(){
    double x;
    cin >> x;
    //对浮点数只能使用大于、小于号进行比较，不能使用等号和不等号，我这里直接把 double 转成字符串，就可以用不等号比较了
    //根据输入条件的限制，可以这样写：x>0.01。
    while(double2Str(x)!="0"){
        int i=2;
        double sum=1.0/i;
        while(sum < x){
            i++;
            sum += 1.0/i;
        }
        cout << i-1 << " card(s)" << endl;
        cin >> x;
    }
    return 0;
}
```
